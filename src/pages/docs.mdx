import Link from '@docusaurus/Link';
import { versions, latestStable } from '@site/src/components/versions.js';

export const VersionList = ({children}) => {
  const VersionLink = (version) => <Link to={`/docs/${version}`}>{version}</Link>;
  return versions.map((version) => <li key={version}>{
    version == latestStable ? [<b>{VersionLink(version)}</b>, ' (latest stable)'] : VersionLink(version)
  }</li>);
}
